// go · testing

go test — тестирование в Go

Полный разбор инструментов тестирования: от базовых команд до табличных тестов, бенчмарков и моков без сторонних библиотек.

01 · basics

Go поставляется со встроенным тестовым фреймворком: пакет testing и команда go test. Тесты живут в файлах с суффиксом _test.go, функции начинаются с Test. Никаких внешних зависимостей не нужно.

go test ./... Запустить все тесты рекурсивно во всех пакетах
go test -v ./... Verbose — показывает имя каждого теста и его статус
go test -run TestName ./... Запустить только тесты, чьё имя соответствует regexp
go test -count=1 ./... Отключить кеширование — count=1 форсирует перезапуск
go test -timeout 30s ./... Установить таймаут (по умолчанию 10 минут) 
// math_test.go — тест-файл для пакета math
package math

import (
    "testing"
)

// Имя: Test + ЧтоПроверяем
func TestAdd(t *testing.T) {
    got := Add(2, 3)
    want := 5

    if got != want {
        t.Errorf("Add(2, 3) = %d, want %d", got, want)
    }
}

// TestMain — setup/teardown перед всеми тестами пакета
func TestMain(m *testing.M) {
    // setup...
    code := m.Run()
    // teardown...
    os.Exit(code)
}
package math

import "testing"

func TestDivide(t *testing.T) {
    result, err := Divide(10.0, 2.0)
    if err != nil {
        t.Fatalf("unexpected error: %v", err) // Fatalf: останавливает тест
    }
    if result != 5.0 {
        t.Errorf("got %f, want 5.0", result) // Errorf: продолжает тест
    }
}

func TestDivideByZero(t *testing.T) {
    _, err := Divide(10.0, 0.0)
    if err == nil {
        t.Error("expected error, got nil")
    }
}

02 · table-driven tests

Идиоматический Go-паттерн: тест-кейсы описываются как срез структур и проходятся в цикле. Убирает дублирование и позволяет добавлять новые случаи одной строкой. В связке с t.Run каждый кейс становится именованным подтестом. 

func TestAdd(t *testing.T) {
    tests := []struct {
        name string
        a, b int
        want int
    }{
        {"positive", 2, 3, 5},
        {"negative", -1, -2, -3},
        {"zero", 0, 0, 0},
        {"mixed", -5, 10, 5},
    }

    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            got := Add(tt.a, tt.b)
            if got != tt.want {
                t.Errorf("Add(%d, %d) = %d, want %d",
                    tt.a, tt.b, got, tt.want)
            }
        })
    }
}
func TestDivide(t *testing.T) {
    tests := []struct {
        name    string
        a, b    float64
        want    float64
        wantErr bool
    }{
        {"normal", 10, 2, 5, false},
        {"by zero", 10, 0, 0, true},
        {"negative", -6, 2, -3, false},
    }

    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            got, err := Divide(tt.a, tt.b)

            if (err != nil) != tt.wantErr {
                t.Fatalf("err = %v, wantErr %v", err, tt.wantErr)
            }
            if !tt.wantErr && got != tt.want {
                t.Errorf("got %v, want %v", got, tt.want)
            }
        })
    }
}
// Явные имена полей — лучше читается при большой таблице
func TestParseDate(t *testing.T) {
    tests := []struct {
        name    string
        input   string
        want    time.Time
        wantErr bool
    }{
        {
            name:  "valid ISO",
            input: "2024-01-15",
            want:  time.Date(2024, time.January, 15, 0, 0, 0, 0, time.UTC),
        },
        {
            name:    "invalid format",
            input:   "15/01/2024",
            wantErr: true,
        },
        {
            name:    "empty",
            input:   "",
            wantErr: true,
        },
    }

    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            got, err := ParseDate(tt.input)
            if (err != nil) != tt.wantErr {
                t.Fatalf("wantErr=%v, got err=%v", tt.wantErr, err)
            }
            if !tt.wantErr && !got.Equal(tt.want) {
                t.Errorf("got %v, want %v", got, tt.want)
            }
        })
    }
}
03 · subtests & t.Parallel

t.Run создаёт подтест с отдельным именем и изолированным состоянием. Подтесты можно фильтровать через -run по иерархическому имени вида TestParent/SubName.

t.Parallel() сигнализирует, что подтест может выполняться параллельно с другими. До Go 1.22 переменную цикла нужно было копировать вручную (tt := tt), чтобы замыкание захватило правильное значение. 

func TestUserService(t *testing.T) {
    svc := NewUserService()

    t.Run("Create", func(t *testing.T) {
        user, err := svc.Create("alice")
        if err != nil {
            t.Fatalf("Create: %v", err)
        }
        if user.Name != "alice" {
            t.Errorf("Name = %q, want %q", user.Name, "alice")
        }
    })

    t.Run("Get/existing", func(t *testing.T) {
        _, err := svc.Get(1)
        if err != nil {
            t.Fatalf("Get: %v", err)
        }
    })

    t.Run("Get/notfound", func(t *testing.T) {
        _, err := svc.Get(999)
        if err == nil {
            t.Error("expected ErrNotFound")
        }
    })
}
func TestHeavyOps(t *testing.T) {
    tests := []struct {
        name  string
        input int
        want  int
    }{
        {"small", 10, 100},
        {"medium", 100, 10000},
        {"large", 1000, 1000000},
    }

    for _, tt := range tests {
        tt := tt // копируем до t.Parallel() (нужно до Go 1.22)
        t.Run(tt.name, func(t *testing.T) {
            t.Parallel() // этот подтест может идти параллельно

            got := Square(tt.input)
            if got != tt.want {
                t.Errorf("Square(%d) = %d, want %d",
                    tt.input, got, tt.want)
            }
        })
    }
}

go test -v -run "^TestUser" ./...
04 · benchmarks

Бенчмарки — функции с сигнатурой BenchmarkXxx(b *testing.B). Go сам подбирает b.N — сколько раз запустить тело цикла для получения стабильного результата. Запускаются только при явном флаге -bench. 

package math

import "testing"

func BenchmarkAdd(b *testing.B) {
    // b.N определяется Go автоматически
    for i := 0; i < b.N; i++ {
        Add(2, 3)
    }
}

func BenchmarkStringConcat(b *testing.B) {
    b.ReportAllocs() // включить отчёт по аллокациям
    for i := 0; i < b.N; i++ {
        s := ""
        for j := 0; j < 100; j++ {
            s += "x"
        }
        _ = s
    }
}

func BenchmarkStringBuilder(b *testing.B) {
    b.ReportAllocs()
    for i := 0; i < b.N; i++ {
        var sb strings.Builder
        for j := 0; j < 100; j++ {
            sb.WriteString("x")
        }
        _ = sb.String()
    }
}
# Все бенчмарки в пакете
go test -bench=. ./...

# Конкретный бенчмарк
go test -bench=BenchmarkStringConcat ./...

# С метриками памяти
go test -bench=. -benchmem ./...

05 · test helpers & t.Helper()

Хелперы — обычные функции, принимающие *testing.T и инкапсулирующие повторяющиеся проверки. Вызов t.Helper() указывает Go, что при ошибке нужно показывать строку вызова хелпера, а не строку внутри него. Без этого дебажить провальные тесты значительно сложнее. 

// ❌ Без t.Helper() — ошибка укажет на строку внутри assertEqual

func assertEqual(t *testing.T, got, want interface{}) {
    // нет t.Helper()
    if got != want {
        t.Errorf("got %v, want %v", got, want)
        // ↑ ошибка покажет эту строку — бесполезно
    }
}

func TestSomething(t *testing.T) {
    assertEqual(t, Add(1, 2), 4)
    // ↑ хотим видеть эту строку при провале
}

// Вывод:
//   helpers_test.go:4: got 3, want 4   ← строка в хелпере, не в тесте
// ✅ С t.Helper() — ошибка укажет на строку вызова

func assertEqual[T comparable](t *testing.T, got, want T) {
    t.Helper() // вот и всё изменение
    if got != want {
        t.Errorf("got %v, want %v", got, want)
    }
}

func assertNoError(t *testing.T, err error) {
    t.Helper()
    if err != nil {
        t.Fatalf("unexpected error: %v", err)
    }
}

func TestSomething(t *testing.T) {
    result, err := Divide(1, 2)
    assertNoError(t, err)
    assertEqual(t, result, 4)
    // ↑ теперь ошибка укажет именно на эту строку
}

// Вывод:
//   something_test.go:14: got 3, want 4   ← строка в тесте
// Хелпер для создания тестовых данных
func mustCreateUser(t *testing.T, svc *UserService, name string) *User {
    t.Helper()
    user, err := svc.Create(name)
    if err != nil {
        t.Fatalf("mustCreateUser: %v", err)
    }
    return user
}

// t.Cleanup — регистрирует teardown, вызывается после теста
func setupDB(t *testing.T) *sql.DB {
    t.Helper()
    db, err := sql.Open("postgres", testDSN)
    if err != nil {
        t.Fatalf("open db: %v", err)
    }
    t.Cleanup(func() { db.Close() }) // автоматический teardown
    return db
}

// Использование
func TestOrder(t *testing.T) {
    db := setupDB(t) // db.Close() вызовется сам
    user := mustCreateUser(t, NewUserService(db), "alice")
    _ = user
}
06 · моки через интерфейсы

В Go не нужны тяжёлые mock-фреймворки для базовых случаев. Интерфейсы — это встроенный механизм мокирования: опишите зависимость как интерфейс, создайте тестовую реализацию, передайте её в конструктор. Реальный код зависит от интерфейса, а не от конкретного типа. 

// Описываем зависимость как интерфейс
type EmailSender interface {
    Send(to, subject, body string) error
}

// Сервис принимает интерфейс, не конкретный тип
type NotificationService struct {
    email EmailSender
}

func NewNotificationService(e EmailSender) *NotificationService {
    return &NotificationService{email: e}
}

func (s *NotificationService) NotifyUser(to, msg string) error {
    return s.email.Send(to, "Уведомление", msg)
}

// Продакшн-реализация
type SMTPSender struct{ host string }

func (s *SMTPSender) Send(to, subj, body string) error {
    // реальная отправка через SMTP...
    return nil
}
// mock_test.go — тестовая реализация EmailSender
type mockEmailSender struct {
    sent []sentEmail
    err  error // если задан — Send вернёт его
}

type sentEmail struct {
    to, subject, body string
}

func (m *mockEmailSender) Send(to, subj, body string) error {
    if m.err != nil {
        return m.err
    }
    m.sent = append(m.sent, sentEmail{to, subj, body})
    return nil
}

// Compile-time проверка что мок реализует интерфейс
var _ EmailSender = (*mockEmailSender)(nil)
func TestNotifyUser(t *testing.T) {
    t.Run("success", func(t *testing.T) {
        mock := &mockEmailSender{}
        svc := NewNotificationService(mock)

        err := svc.NotifyUser("user@example.com", "hello")
        if err != nil {
            t.Fatalf("unexpected error: %v", err)
        }
        if len(mock.sent) != 1 {
            t.Fatalf("expected 1 sent email, got %d", len(mock.sent))
        }
        if mock.sent[0].to != "user@example.com" {
            t.Errorf("wrong recipient: %s", mock.sent[0].to)
        }
    })

    t.Run("sender error", func(t *testing.T) {
        mock := &mockEmailSender{err: errors.New("smtp down")}
        svc := NewNotificationService(mock)

        err := svc.NotifyUser("user@example.com", "hello")
        if err == nil {
            t.Error("expected error")
        }
    })
}
07 · coverage

Go умеет измерять покрытие без внешних инструментов. 100% coverage не означает отсутствие багов — это лишь метрика того, что каждая строчка была выполнена. Фокусируйтесь на покрытии граничных случаев и путей ошибок.

go test -cover ./... Показать процент покрытия для каждого пакета
go test -coverprofile=c.out ./... Сохранить детальный профиль покрытия в файл
go tool cover -html=c.out HTML-отчёт с подсветкой: зелёный — покрыто, красный — нет
go tool cover -func=c.out Покрытие по каждой функции в терминале
