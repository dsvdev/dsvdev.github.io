Go context.Context

Стандартный механизм управления жизненным циклом операций в Go. Отменяет запросы, задаёт дедлайны и передаёт данные между горутинами без засорения сигнатур функций.

зачем нужен context
🚫
Отмена операций
Клиент закрыл соединение — незачем продолжать вычисления. ctx.Done() сигнализирует всем дочерним горутинам одновременно.
⏱
Дедлайны и таймауты
Внешний сервис отвечает слишком долго. WithTimeout гарантирует, что запрос не зависнет навсегда.
🔗
Передача значений
Request-scoped данные (trace ID, userID) передаются через весь стек вызовов без изменения каждой функции.

Каждый HTTP-запрос порождает горутину, которая вызывает БД, которая вызывает кэш, которая вызывает внешний API. Если клиент отключился — все эти горутины продолжают работать впустую.

context.Context — это дерево отмены. Когда родительский контекст отменяется, все дочерние получают сигнал через канал Done(). Каждая функция в цепочке должна слушать этот канал и корректно завершаться.

context.Background() и context.TODO()

context.Background() — пустой корневой контекст без дедлайна, отмены и значений. Используй в точках входа: main(), обработчиках HTTP-запросов, тестах.

context.TODO() семантически идентичен Background, но сигнализирует: «я пока не знаю, какой контекст здесь нужен». Полезен при рефакторинге — grep по TODO() находит места, которые ещё не доработаны. 

func main() {
    // точка входа — всегда Background
    ctx := context.Background()
    runServer(ctx)
}

func RequestIDMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        // r.Context() уже несёт отмену при разрыве соединения
        ctx := context.WithValue(r.Context(), reqIDKey, r.Header.Get("X-Request-ID"))
        next.ServeHTTP(w, r.WithContext(ctx))
    })
}

func legacyFn() {
    // TODO: добавить ctx когда будем рефакторить этот слой
    ctx := context.TODO()
    doWork(ctx)
}
context.WithCancel

Возвращает дочерний контекст и функцию cancel(). Вызов cancel() закрывает канал ctx.Done() — все горутины, ждущие этого сигнала, получают его одновременно.

Основной паттерн: запускаешь несколько горутин, в одной приходит ошибка — отменяешь все остальные одним вызовом cancel(). 

package main

import (
    "context"
    "fmt"
    "time"
)

func worker(ctx context.Context, id int) {
    for {
        select {
        case <-ctx.Done():
            fmt.Printf("worker %d остановлен: %v\n", id, ctx.Err())
            return
        default:
            fmt.Printf("worker %d работает...\n", id)
            time.Sleep(500 * time.Millisecond)
        }
    }
}

func main() {
    ctx, cancel := context.WithCancel(context.Background())
    defer cancel() // всегда defer cancel — иначе goroutine leak

    go worker(ctx, 1)
    go worker(ctx, 2)

    time.Sleep(2 * time.Second)
    cancel() // оба worker'а получат сигнал и остановятся
    time.Sleep(100 * time.Millisecond)
}
⚠ подводный камень
Всегда вызывай cancel() — даже если контекст уже истёк сам по себе. Иначе runtime удерживает внутренние таймеры и ресурсы до следующей GC. Паттерн defer cancel() сразу после создания контекста полностью решает эту проблему.
context.WithTimeout

Автоматически отменяет контекст через указанный промежуток времени. Идеален для HTTP-клиентов, запросов к БД, вызовов внешних сервисов — везде, где нельзя ждать бесконечно.

http.NewRequestWithContext — правильный способ привязать контекст к HTTP-запросу. Если сервер не ответит вовремя, запрос будет отменён с ошибкой context.DeadlineExceeded. 

package main

import (
    "context"
    "fmt"
    "net/http"
    "time"
)

func fetchUser(userID string) (int, error) {
    ctx, cancel := context.WithTimeout(context.Background(), 3*time.Second)
    defer cancel()

    url := fmt.Sprintf("https://api.example.com/users/%s", userID)
    req, err := http.NewRequestWithContext(ctx, http.MethodGet, url, nil)
    if err != nil {
        return 0, err
    }

    resp, err := http.DefaultClient.Do(req)
    if err != nil {
        // если таймаут — err обернёт context.DeadlineExceeded
        return 0, fmt.Errorf("fetch user: %w", err)
    }
    defer resp.Body.Close()

    return resp.StatusCode, nil
}
⚠ подводный камень
Не устанавливай таймаут через http.Client.Timeout и context.WithTimeout одновременно без понимания — сработает тот, что наступит раньше. Для таймаута конкретного запроса используй контекст; Client.Timeout — глобальная защита всего клиента.
context.WithDeadline

WithDeadline принимает абсолютное время, WithTimeout — относительное. Под капотом WithTimeout(ctx, d) — это просто WithDeadline(ctx, time.Now().Add(d)).

Используй WithDeadline, когда дедлайн задан снаружи — например, как часть SLA всей операции. Можно «пробросить» дедлайн вниз, не добавляя нового. 

func processWithSLA(ctx context.Context) error {
    // проверяем, есть ли дедлайн у родителя
    parentDeadline, ok := ctx.Deadline()
    if !ok {
        // родитель без дедлайна — устанавливаем свой
        var cancel context.CancelFunc
        ctx, cancel = context.WithTimeout(ctx, 5*time.Second)
        defer cancel()
    } else {
        // дедлайн есть — берём тот, что раньше наступит
        ourMax := time.Now().Add(5 * time.Second)
        if parentDeadline.After(ourMax) {
            var cancel context.CancelFunc
            ctx, cancel = context.WithDeadline(ctx, ourMax)
            defer cancel()
        }
        // иначе родительский дедлайн строже — не трогаем
    }

    return doHeavyWork(ctx)
}
совет
ctx.Deadline() возвращает (time.Time, bool). Всегда проверяй bool перед использованием — если дедлайна нет, время будет нулевым и сравнения будут некорректными.
context.WithValue

Прикрепляет к контексту произвольное значение. Предназначен для request-scoped данных: trace ID, correlation ID, authenticated user — данных, нужных всему стеку, но не меняющих логику функции.

Никогда не используй string или встроенные типы как ключи — разные пакеты случайно перезапишут значение друг друга. Всегда объявляй собственный unexported тип ключа. 

package middleware

import (
    "context"
    "net/http"
)

// unexported тип — защита от коллизий между пакетами
type ctxKey string

const requestIDKey ctxKey = "requestID"

func RequestID(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        id := r.Header.Get("X-Request-ID")
        ctx := context.WithValue(r.Context(), requestIDKey, id)
        next.ServeHTTP(w, r.WithContext(ctx))
    })
}

// GetRequestID — типизированный accessor; не разбрасывай ctx.Value по коду
func GetRequestID(ctx context.Context) string {
    id, _ := ctx.Value(requestIDKey).(string)
    return id
}
⚠ подводный камень
WithValue — не замена конфигурации и не кэш. Не храни в контексте зависимости (logger, db, repo) — это нарушает явность и ломает тестируемость. Контекст для данных конкретного запроса (кто, откуда, с каким trace), а не для того, что нужно функции для работы.
propagation через горутины

context.Context должен быть первым параметром каждой функции, выполняющей I/O или способной занять продолжительное время. Это конвенция Go.

При запуске горутины — передавай тот же контекст в замыкание. Создавать новый контекст внутри горутины — антипаттерн: она выживет дольше, чем должна, после отмены родителя. 

func processOrder(ctx context.Context, orderID int64) error {
    order, err := fetchOrder(ctx, orderID) // ctx прокидываем везде
    if err != nil {
        return err
    }

    errCh := make(chan error, 2)

    go func() {
        // ctx захвачен до старта горутины — корректно
        errCh <- updateInventory(ctx, order)
    }()

    go func() {
        errCh <- sendConfirmation(ctx, order)
    }()

    for i := 0; i < 2; i++ {
        if err := <-errCh; err != nil {
            return err
        }
    }
    return nil
}
правило 1
Не храни контекст в полях структуры — передавай явно в каждый метод. — Исключение: контекст, чей срок жизни совпадает со сроком жизни объекта.
правило 2
Никогда не передавай nil в качестве контекста. Если не знаешь что передать — context.TODO(). — nil вызовет панику в первом же ctx.Done() или ctx.Err().
правило 3
Получил контекст снаружи — не создавай новый таймаут поверх него без проверки. Иначе можешь сделать дедлайн мягче, чем у вызывающего. — Используй ctx.Deadline() чтобы выбрать меньшее из двух значений.
cheatsheet — что выбрать
ситуация API примечание
Начало цепочки (main, тест, обработчик) context.Background() всегда корень дерева контекстов
Место будет заменено при рефакторинге context.TODO() grep-маркер незавершённых мест
Нужно отменить операцию вручную context.WithCancel() всегда defer cancel()
Операция должна завершиться за N секунд context.WithTimeout() удобнее WithDeadline для относительного времени
Операция должна завершиться к конкретному моменту context.WithDeadline() абсолютное время; удобно для SLA всей операции
Передать request-scoped данные (trace ID, userID) context.WithValue() ключ — unexported тип; accessor — отдельная функция
