Go Sync Primitives

Конкуренция в Go — это сила. Sync-примитивы — это узда. Без узды лошадь убьёт тебя. Разбираем каждый инструмент: когда брать, когда не трогать и где он укусит.

sync.Mutex
🔒
sync.Mutex
Когда: общий ресурс, один писатель / один читатель, простая защита критической секции.
mutex_counter.go Go 
type Counter struct {
    mu    sync.Mutex
    value int64
}

func (c *Counter) Inc() {
    c.mu.Lock()
    defer c.mu.Unlock()
    c.value++
}

func (c *Counter) Value() int64 {
    c.mu.Lock()
    defer c.mu.Unlock()
    return c.value
}
⚠ Подводные камни
Никогда не копируй sync.Mutex после первого использования — vet поймает, но не всегда. Передавай структуру через указатель. Не блокируй повторно из той же горутины — дедлок гарантирован. Держи мьютекс как можно короче; не делай под ним сетевых вызовов.
sync.RWMutex
📖
sync.RWMutex
Когда: много читателей, мало писателей. Read-heavy кэши, конфиги, справочники.
rwmutex_cache.go Go 
type Cache struct {
    mu   sync.RWMutex
    data map[string]string
}

func (c *Cache) Get(key string) (string, bool) {
    c.mu.RLock()           // разделяемая блокировка — читатели не мешают друг другу
    defer c.mu.RUnlock()
    v, ok := c.data[key]
    return v, ok
}

func (c *Cache) Set(key, value string) {
    c.mu.Lock()            // эксклюзивная блокировка — все ждут
    defer c.mu.Unlock()
    c.data[key] = value
}
⚠ Подводные камни
RWMutex медленнее Mutex при высоком contention из-за накладных расходов на управление читателями. Если записей столько же, сколько чтений — бери обычный Mutex. Никогда не делай апгрейд RLock → Lock без RUnlock — это дедлок.
sync.WaitGroup
⏳
sync.WaitGroup
Когда: нужно дождаться завершения N горутин. Fan-out, параллельные запросы, worker pool.
waitgroup_fanout.go Go 
func fetchAll(urls []string) []string {
    var (
        wg      sync.WaitGroup
        mu      sync.Mutex
        results []string
    )

    for _, url := range urls {
        wg.Add(1)
        go func(u string) {
            defer wg.Done()
            body := fetch(u) // ваш http.Get
            mu.Lock()
            results = append(results, body)
            mu.Unlock()
        }(url)
    }

    wg.Wait()
    return results
}
⚠ Подводные камни
wg.Add(1) вызывай до старта горутины, не внутри неё — гонка гарантирована. Не передавай WaitGroup по значению — только указатель или закрытие через переменную. Счётчик никогда не должен уходить в минус: Done() без Add() — паника.
sync.Once
1️⃣
sync.Once
Когда: ленивая инициализация, синглтон, однократное выполнение дорогой операции.
once_singleton.go Go 
type DB struct {
    once sync.Once
    conn *sql.DB
}

func (d *DB) Conn() *sql.DB {
    d.once.Do(func() {
        var err error
        d.conn, err = sql.Open("postgres", dsn)
        if err != nil {
            panic(err) // Once не повторится даже при ошибке
        }
    })
    return d.conn
}

// Все вызовы после первого — мгновенны, без блокировки
⚠ Подводные камни
Once не повторяется даже если функция вернула ошибку или запаниковала (с Go 1.14 паника всё же пробрасывается). Если инициализация провалилась — Do больше не вызовется. Обрабатывай ошибки через sync.OnceValue (Go 1.21+) или храни err отдельно.
sync.Cond
📡
sync.Cond
Когда: горутина должна ждать выполнения условия, а не просто сигнала. Очереди, producer-consumer, буфер с дроп-логикой.
cond_queue.go Go 
type Queue struct {
    mu    sync.Mutex
    cond  *sync.Cond
    items []int
}

func NewQueue() *Queue {
    q := &Queue{}
    q.cond = sync.NewCond(&q.mu)
    return q
}

func (q *Queue) Put(v int) {
    q.mu.Lock()
    q.items = append(q.items, v)
    q.cond.Signal() // будим одного ожидающего
    q.mu.Unlock()
}

func (q *Queue) Get() int {
    q.mu.Lock()
    defer q.mu.Unlock()
    for len(q.items) == 0 {
        q.cond.Wait() // атомарно снимает мьютекс и засыпает
    }
    v := q.items[0]
    q.items = q.items[1:]
    return v
}
⚠ Подводные камни
Wait() всегда вызывай в for-цикле — spurious wakeups реальны. Нельзя использовать Cond со значением мьютекса — только указатель. В большинстве случаев channel проще и читабельнее. Cond нужен, когда одним каналом не обойтись или нужен Broadcast.
sync/atomic
⚡
sync/atomic
Когда: простые счётчики, флаги, hot path где мьютекс слишком дорог. Lock-free структуры.
atomic_counter.go Go 
import "sync/atomic"

// Go 1.19+: типизированные атомики
var requests atomic.Int64

func handler(w http.ResponseWriter, r *http.Request) {
    requests.Add(1)
    // ...
}

func stats() int64 {
    return requests.Load()
}

// CAS — compare-and-swap: меняет только если значение совпало
var flag atomic.Bool

func tryAcquire() bool {
    return flag.CompareAndSwap(false, true)
}
⚠ Подводные камни
Атомики работают только для одиночных операций. Если нужно обновить два поля одновременно — только Mutex. atomic.Value требует, чтобы тип хранимого значения не менялся — паника при несовпадении типов. Не решай атомиком задачи мьютекса: "горящие флаги" вместо нормальной синхронизации — путь к race condition.
channels как sync-примитив
🔗
channel
Когда: передача владения данными, оркестрация горутин, сигналы завершения, pipeline, таймауты.
channel_patterns.go Go 
// ── Сигнал завершения (done channel) ──
func worker(done <-chan struct{}) {
    for {
        select {
        case <-done:
            return // получили сигнал — выходим
        default:
            doWork()
        }
    }
}

done := make(chan struct{})
go worker(done)
close(done) // будит всех читателей сразу

// ── Семафор через буферизованный канал ──
sem := make(chan struct{}, 5) // не более 5 горутин

for _, task := range tasks {
    sem <- struct{}{}
    go func(t Task) {
        defer func() { <-sem }()
        process(t)
    }(task)
}
⚠ Подводные камни
Запись в закрытый канал — паника. Читай из закрытого — всегда получишь нулевое значение. Небуферизованный канал блокирует оба конца до встречи — легко получить дедлок. Не передавай канал туда и обратно — определи единственного владельца и закрывай только его стороной.
что выбрать
Примитив Используй когда Не используй когда
Mutex Нужна простая защита разделяемого состояния Много читателей → RWMutex; счётчики → atomic
RWMutex Read >> Write, read-heavy кэши Запись столь же часта, как чтение
WaitGroup Ждать N горутин, fan-out Нужно получить результат — лучше канал
Once Ленивая инициализация, синглтон Нужно повторить при ошибке
Cond Ожидание условия, Broadcast нескольким Большинство случаев — канал читабельнее
atomic Счётчики, флаги, hot path Составные операции над несколькими полями
channel Передача данных/сигналов между горутинами Разделяемое состояние — используй Mutex
